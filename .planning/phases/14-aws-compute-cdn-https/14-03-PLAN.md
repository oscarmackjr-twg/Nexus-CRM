**terraform/modules/cloudfront/main.tf:**
```hcl
locals {
  name_prefix = "${var.app_name}-${var.environment}"
}

# --- S3 Frontend Bucket ---
# Static website hosting is NOT enabled. CloudFront reads via OAC using the
# S3 REST API endpoint (bucket_regional_domain_name), not the website endpoint.
# SPA 404 routing is handled entirely by CloudFront custom_error_response.

resource "aws_s3_bucket" "frontend" {
  bucket = "${local.name_prefix}-frontend"

  tags = {
    Name = "${local.name_prefix}-frontend"
  }
}

resource "aws_s3_bucket_public_access_block" "frontend" {
  bucket                  = aws_s3_bucket.frontend.id
  block_public_acls       = true
  block_public_policy     = true
  ignore_public_acls      = true
  restrict_public_buckets = true
}

resource "aws_s3_bucket_versioning" "frontend" {
  bucket = aws_s3_bucket.frontend.id

  versioning_configuration {
    status = "Enabled"
  }
}

# --- CloudFront Origin Access Control ---
# OAC preferred over legacy OAI. OAC uses SigV4 signing and is the current
# AWS standard. Do NOT use aws_cloudfront_origin_access_identity (OAI).

resource "aws_cloudfront_origin_access_control" "frontend" {
  name                              = "${local.name_prefix}-frontend-oac"
  origin_access_control_origin_type = "s3"
  signing_behavior                  = "always"
  signing_protocol                  = "sigv4"
}

# --- CloudFront Distribution ---

resource "aws_cloudfront_distribution" "this" {
  enabled             = true
  is_ipv6_enabled     = true
  default_root_object = "index.html"
  aliases             = [var.app_domain]
  price_class         = "PriceClass_200"  # US, Canada, Europe, Asia, Middle East, Africa

  # Origin 1: S3 (for React SPA static assets — default origin)
  origin {
    domain_name              = aws_s3_bucket.frontend.bucket_regional_domain_name
    origin_id                = "S3-${local.name_prefix}-frontend"
    origin_access_control_id = aws_cloudfront_origin_access_control.frontend.id
  }

  # Origin 2: ALB (for /api/* and /health* path behaviors)
  origin {
    domain_name = var.alb_dns_name
    origin_id   = "ALB-${local.name_prefix}"

    custom_origin_config {
      http_port              = 80
      https_port             = 443
      origin_protocol_policy = "https-only"
      origin_ssl_protocols   = ["TLSv1.2"]
    }
  }

  # Default cache behavior: serve from S3 (React SPA)
  default_cache_behavior {
    target_origin_id       = "S3-${local.name_prefix}-frontend"
    viewer_protocol_policy = "redirect-to-https"
    allowed_methods        = ["GET", "HEAD", "OPTIONS"]
    cached_methods         = ["GET", "HEAD"]
    compress               = true

    forwarded_values {
      query_string = false
      cookies { forward = "none" }
    }

    min_ttl     = 0
    default_ttl = 3600
    max_ttl     = 86400
  }

  # /api/* behavior: proxy to ALB with no caching
  # Authorization header MUST be forwarded — CloudFront strips it by default.
  # Failure to forward Authorization causes 401 responses at the API even when
  # the client sends a valid Bearer token.
  ordered_cache_behavior {
    path_pattern           = "/api/*"
    target_origin_id       = "ALB-${local.name_prefix}"
    viewer_protocol_policy = "https-only"
    allowed_methods        = ["DELETE", "GET", "HEAD", "OPTIONS", "PATCH", "POST", "PUT"]
    cached_methods         = ["GET", "HEAD"]
    compress               = false

    forwarded_values {
      query_string = true
      headers      = ["Authorization", "Content-Type", "X-Request-ID", "Origin", "Accept"]
      cookies { forward = "all" }
    }

    # No caching for API responses
    min_ttl     = 0
    default_ttl = 0
    max_ttl     = 0
  }

  # /health* behavior: proxy to ALB (health checks, monitoring, Phase 14 smoke tests)
  ordered_cache_behavior {
    path_pattern           = "/health*"
    target_origin_id       = "ALB-${local.name_prefix}"
    viewer_protocol_policy = "https-only"
    allowed_methods        = ["GET", "HEAD"]
    cached_methods         = ["GET", "HEAD"]
    compress               = false

    forwarded_values {
      query_string = false
      cookies { forward = "none" }
    }

    min_ttl     = 0
    default_ttl = 0
    max_ttl     = 0
  }

  # SPA routing: S3 returns 403 for missing keys (bucket is private, not 404).
  # Both 403 and 404 must be mapped to /index.html so React Router handles the path.
  custom_error_response {
    error_code            = 403
    response_code         = 200
    response_page_path    = "/index.html"
    error_caching_min_ttl = 0
  }

  custom_error_response {
    error_code            = 404
    response_code         = 200
    response_page_path    = "/index.html"
    error_caching_min_ttl = 0
  }

  restrictions {
    geo_restriction {
      restriction_type = "none"
    }
  }

  viewer_certificate {
    acm_certificate_arn      = var.cloudfront_certificate_arn  # us-east-1 cert — see variable description
    ssl_support_method       = "sni-only"
    minimum_protocol_version = "TLSv1.2_2021"
  }

  tags = {
    Name = "${local.name_prefix}-cloudfront"
  }
}

# --- S3 Bucket Policy: Allow CloudFront OAC ---
# The bucket policy references aws_cloudfront_distribution.this.arn which creates
# an implicit Terraform dependency. Terraform will create the distribution first,
# then apply the bucket policy. This is NOT a circular dependency.

data "aws_iam_policy_document" "frontend_bucket" {
  statement {
    sid     = "AllowCloudFrontServicePrincipal"
    effect  = "Allow"
    actions = ["s3:GetObject"]
    resources = ["${aws_s3_bucket.frontend.arn}/*"]

    principals {
      type        = "Service"
      identifiers = ["cloudfront.amazonaws.com"]
    }

    condition {
      test     = "StringEquals"
      variable = "AWS:SourceArn"
      values   = [aws_cloudfront_distribution.this.arn]
    }
  }
}

resource "aws_s3_bucket_policy" "frontend" {
  bucket = aws_s3_bucket.frontend.id
  policy = data.aws_iam_policy_document.frontend_bucket.json

  depends_on = [aws_s3_bucket_public_access_block.frontend]
}

# --- Route 53: crm.oscarmackjr.com → CloudFront ---
# Single entry point: all traffic (SPA + API) enters through CloudFront.
# CloudFront routes /api/* to ALB internally via path behavior.
# CloudFront distributions always use hosted_zone_id = "Z2FDTNDATAQYW2" for
# Route 53 alias records — this is an AWS constant, not a user-defined value.

resource "aws_route53_record" "app" {
  zone_id = var.route53_zone_id
  name    = var.app_domain
  type    = "A"

  alias {
    name                   = aws_cloudfront_distribution.this.domain_name
    zone_id                = aws_cloudfront_distribution.this.hosted_zone_id
    evaluate_target_health = false
  }
}
```
