```jsx
import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useQuery } from '@tanstack/react-query';
import { getContacts } from '@/api/contacts';
import { DataGrid } from '@/components/DataGrid';

import { Input } from '@/components/ui/input';
import { formatDate } from '@/lib/utils';
```

State: `const [page, setPage] = useState(1)`, `const [size, setSize] = useState(25)`,
`const [search, setSearch] = useState('')`

Query (TanStack Query — see version note in interfaces above):
```jsx
const { data, isLoading } = useQuery({
  queryKey: ['contacts', { page, size, search }],
  queryFn: () => getContacts({ page, size, ...(search ? { search } : {}) }),
  keepPreviousData: true,  // v4 — swap for placeholderData: keepPreviousData if v5
});
```

Handle size change: `setSize(n); setPage(1)` — reset to page 1 when size changes.
Handle search change: `setSearch(val); setPage(1)` — reset to page 1 on new search.

return (
  <div className="space-y-4">
