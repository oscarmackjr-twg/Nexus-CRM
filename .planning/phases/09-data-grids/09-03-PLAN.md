**DealsPage.jsx full implementation:**

```jsx
import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useQuery } from '@tanstack/react-query';
import { getDeals } from '@/api/deals';
import { DataGrid } from '@/components/DataGrid';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { formatCurrency, formatDate } from '@/lib/utils';
import { cn } from '@/lib/utils';
```

State:
```jsx
const [page, setPage] = useState(1);
const [size, setSize] = useState(25);
const [search, setSearch] = useState('');
const [status, setStatus] = useState('');  // '' = All, 'open', 'won', 'lost'
const navigate = useNavigate();
```

Query:
```jsx
const { data, isLoading } = useQuery({
  queryKey: ['deals', { page, size, search, status }],
  queryFn: () => getDeals({
    page,
    size,
    ...(search ? { search } : {}),
    ...(status ? { status } : {}),
  }),
  keepPreviousData: true,  // v4 — swap for placeholderData: keepPreviousData if v5
});
```

**Status badge helper:**
```jsx
function StatusBadge({ status }) {
  const variants = {
    open: 'bg-blue-100 text-blue-700',
    won:  'bg-green-100 text-green-700',
    lost: 'bg-gray-100 text-gray-600',
  };
  return (
    <span className={cn('text-xs px-2 py-0.5 rounded-full font-medium capitalize', variants[status] ?? 'bg-gray-100 text-gray-600')}>
      {status ?? '—'}
    </span>
  );
}
```

```jsx
      )}
    >
      {f.label}
    </button>
  ))}
</div>
```

**Full JSX return:**
```jsx
return (
  <div className="space-y-4">
    <div className="flex items-center justify-between">

      <Input
        placeholder="Search deals..."

        className="w-64 text-sm"

          className={cn(

            status === f.value
              ? 'bg-[#1a3868] text-white border-[#1a3868]'

          {f.label}
```
